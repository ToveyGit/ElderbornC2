ek_race_compatible_trigger = {
	trigger_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_khajiit }
				$CHARACTER2$ = { has_character_flag = race_khajiit }
			}
		}
		$CHARACTER1$ = { has_character_flag = race_khajiit }
		$CHARACTER2$ = { has_character_flag = race_khajiit }
	}
	trigger_else_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_argonian }
				$CHARACTER2$ = { has_character_flag = race_argonian }
			}
		}
		$CHARACTER1$ = { has_character_flag = race_argonian }
		$CHARACTER2$ = { has_character_flag = race_argonian }
	}
	trigger_else_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_lilmothiit }
				$CHARACTER2$ = { has_character_flag = race_lilmothiit }
			}
		}
		$CHARACTER1$ = { has_character_flag = race_lilmothiit }
		$CHARACTER2$ = { has_character_flag = race_lilmothiit }
	}
	trigger_else_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_imga }
				$CHARACTER2$ = { has_character_flag = race_imga}
			}
		}
		$CHARACTER1$ = { has_character_flag = race_imga }
		$CHARACTER2$ = { has_character_flag = race_imga }
	}
	trigger_else_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_goblinken }
				$CHARACTER2$ = { has_character_flag = race_goblinken }
			}
		}
		$CHARACTER1$ = { has_character_flag = race_goblinken }
		$CHARACTER2$ = { has_character_flag = race_goblinken }
	}
	trigger_else_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_minotaur }
				$CHARACTER2$ = { has_character_flag = race_minotaur }
			}
		}
		$CHARACTER1$ = { has_character_flag = race_minotaur }
		$CHARACTER2$ = { has_character_flag = race_minotaur }
	}
	#trigger_else_if = {
		#limit = {
			#OR = {
				#$CHARACTER1$ = { has_character_flag = race_giant }
				#$CHARACTER2$ = { has_character_flag = race_giant }
			#}
		#}
		#$CHARACTER1$ = { has_character_flag = race_giant }
		#$CHARACTER2$ = { has_character_flag = race_giant }
	#}
	trigger_else_if = {
		limit = {
			OR = {
				$CHARACTER1$ = { has_character_flag = race_daedra }
				$CHARACTER2$ = { has_character_flag = race_daedra }
			}
		}
		$CHARACTER1$ = { has_character_flag = race_daedra }
		$CHARACTER2$ = { has_character_flag = race_daedra }
	}
	trigger_else = {
		always = yes
	}
}
