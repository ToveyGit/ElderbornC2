elsweyr.0002 = {
	title = elsweyr.0002.t
	desc = elsweyr.0002.desc
	content_source = dlc_ek
	theme = realm

	left_portrait = {
		character = root
		animation = personality_bold
	}

	immediate = {
		title:k_leyawiin = { 
			set_title_name = BLACKWOOD
			set_capital_county = title:c_blackwood 
		}
	}

	#Settle Malapiit
	option = {
		name = elsweyr.0002.a
		custom_tooltip = elsweyr.0002.a.tt
		get_title = title:c_leyawiin
		title:c_leyawiin = {
			set_county_culture = culture:malapiit
		}
		culture:nibenese = {
			change_cultural_acceptance = {
				target = root.culture
				value = medium_negative_culture_acceptance
				desc = cultural_acceptance_historical_relations
			}
		}
		culture:malapiit = {
			change_cultural_acceptance = {
				target = root.culture
				value = medium_positive_culture_acceptance
				desc = cultural_acceptance_historical_relations
			}
		}
		hidden_effect = {
			title:d_nomore = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_county}
					add_to_list = malapiit_counties
				}
			}
			title:d_leyawiin = {
				every_in_de_jure_hierarchy = {
					limit = { tier = tier_county}
					add_to_list = malapiit_counties
				}
			}
			every_in_list = {
				list = malapiit_counties
				add_county_modifier = {
					modifier = malapiit_flourishing_modifier
					years = 30
				}
			}				
		}
	}
	
	#Settle your people
	option = {
		name = elsweyr.0002.b
		custom_tooltip = elsweyr.0002.b.tt
		get_title = title:c_leyawiin
        title:c_leyawiin = {
			set_county_culture = root.culture
			set_county_faith = root.faith
		}
		culture:nibenese = {
			change_cultural_acceptance = {
				target = root.culture
				value = medium_negative_culture_acceptance
				desc = cultural_acceptance_historical_relations
			}
		}
		culture:malapiit = {
			change_cultural_acceptance = {
				target = root.culture
				value = low_negative_culture_acceptance
				desc = cultural_acceptance_historical_relations
			}
		}
	}

	#Keep Imperials in Leyawiin
	option = {
		name = elsweyr.0002.c
		custom_tooltip = elsweyr.0002.c.tt
		add_prestige = 500
		add_piety = 500
		culture:nibenese = {
			change_cultural_acceptance = {
				target = root.culture
				value = low_positive_culture_acceptance
				desc = cultural_acceptance_historical_relations
			}
		}
	}
}

#Dissolution event
elsweyr.0005 = {
	title = elsweyr.0005.t
	content_source = dlc_ek
	theme = realm
	override_background = { reference = throne_room_india }
	# which characters to highlight depends on the circumstances

	left_portrait = {
		trigger = {
			exists = scope:kingdom_former
		}
		character = scope:kingdom_former
		animation = personality_bold
	}
	right_portrait = {
		trigger = {
			NOT = { exists = scope:kingdom_former }
		}
		character = scope:elsweyr_holder
		animation = disbelief
	}
	lower_right_portrait = {
		trigger = {
			exists = scope:kingdom_former
		}
		character = scope:elsweyr_holder
	}

	desc = {
		desc = elsweyr.0005.desc.1 #this first paragraph is used for all variants of the event

		#the second one varies depending on why Elsweyr split
		first_valid = {
			# Pellitine formed
			triggered_desc = {
				trigger = {
					is_target_in_global_variable_list = {
						name = unavailable_unique_decisions
						target = flag:pellitine_restored
					} 
				}
				desc = elsweyr.0005.desc.2.pellitine
			}
			# Anequina formed
			triggered_desc = {
				trigger = {
					is_target_in_global_variable_list = {
						name = unavailable_unique_decisions
						target = flag:anequina_restored
					} 
				}
				desc = elsweyr.0005.desc.2.anequina
			}
			desc = elsweyr.0005.desc.2 # for when it just splits by itself
		}
	}
	
	immediate = {
		# for loc purposes, check if Elsweyr currently has a holder
		if = {
			limit = { exists = title:e_elsweyr.holder }
			title:e_elsweyr.holder = { save_scope_as = elsweyr_holder }
			title:e_elsweyr.holder = { add_legitimacy = massive_legitimacy_loss } #Lose legitimacy
		}
		# if not, grab the last one
		else = {
			title:e_elsweyr.previous_holder = { save_scope_as = elsweyr_holder }
		}
		# also for Anequina and Pellitine
		if = {
			limit = { exists = title:e_anequina.holder }
			title:e_anequina.holder = { save_scope_as = kingdom_former }
		}
		if = {
			limit = { exists = title:e_pellitine.holder }
			title:e_pellitine.holder = { save_scope_as = kingdom_former }
		}
	}

	option = {
		name = elsweyr.0005.a
		custom_tooltip = anequina_kingdoms.tt
		custom_tooltip = pellitine_kingdoms.tt
		custom_tooltip = elsweyr_dissolve.create_de_jure_empires.tt
		custom_tooltip = elsweyr_dissolve.become_titular.tt
		custom_tooltip = elsweyr_dissolve.reunite_elsweyr.tt
		flavor = {
			# to give different loc based on the reason why Elsweyr dissolved
			first_valid = {
				# Pellitine formed
				triggered_desc = {
					trigger = {
						is_target_in_global_variable_list = {
							name = unavailable_unique_decisions
							target = flag:pellitine_restored
						} 
					}
					desc = elsweyr.0005.flavor.pellitine
				}
				# Anequina formed
				triggered_desc = {
					trigger = {
						is_target_in_global_variable_list = {
							name = unavailable_unique_decisions
							target = flag:anequina_restored
						} 
					}
					desc = elsweyr.0005.flavor.anequina
				}
				# 20 years passed without anyone holding Elsweyr
				triggered_desc = {
					trigger = {
						title:e_elsweyr = {
							NOT = { is_title_created = yes }
							# NOT = { recent_history = { years = 20 } }
						}
					}
					desc = elsweyr.0005.flavor.no_elsweyr
				}
				# Mane unable to control Elweyr by 2E 490
				triggered_desc = {
					trigger = {
						AND = {
							#Dissolves 100 years after Ridthar-Ridatta if Mane doesn't fully control Elsweyr 
							current_date >= 3410.1.1
							#For testing
							#current_date >= 3372.1.1
							#Mane controls at least 75% of Elsweyr
							any_county_in_region = {
								region = mundus_tamriel_elsweyr
								percent < 0.75
								holder.top_liege = title:e_elsweyr.holder
							}
						}
					}
					desc = elsweyr.0005.flavor.weak_mane
				}
			}
		}
	}
}

						}

							scope:title = title:e_anequina
							title:e_anequina = {
								recent_history = {
									type = conquest_claim
									years = 1
								}
							}
						}
					}
				}
				desc = elsweyr.0006.desc.war
			}
			# Doing it in an other way
			triggered_desc = {
				trigger = {
					NOR = {
						AND = {
							scope:title = title:e_pellitine
							title:e_pellitine = {
								recent_history = {
									type = conquest_claim
									years = 1
								}
							}
						}

							scope:title = title:e_anequina

									years = 1
								}
							}
						}
					}
				}
				desc = elsweyr.0006.desc.other
			}
		}
		first_valid = {
			# Doing it as Pellitine
			triggered_desc = {
				trigger = {
					AND = {
						OR = {
							culture = culture:pellitinian
							culture = { any_parent_culture_or_above = { this = culture:pellitinian } }
						}
					}
					NOT = { has_title = title:d_mane }
				}
				desc = elsweyr.0006.desc.pellitine
			}
			# Doing it as Anequina
			triggered_desc = {
				trigger = {
					AND = {
						OR = {
							culture = culture:anequinan
							culture = { any_parent_culture_or_above = { this = culture:anequinan } }
						}
						NOT = { has_title = title:d_mane }
					}
				}
				desc = elsweyr.0006.desc.anequina
			}
			#Doing it as Baandari
			triggered_desc = {
				trigger = {
					AND = {
						OR = {
							culture = culture:baandari
							culture = { any_parent_culture_or_above = { this = culture:baandari } }
						}
						NOT = { has_title = title:d_mane }
					}
				}
				desc = elsweyr.0006.desc.baandari
			}
			# Doing it as Mane
			triggered_desc = {
				trigger = {
					has_title = title:d_mane
				}
				desc = elsweyr.0006.desc.mane
			}
			triggered_desc = {
				trigger = {
					OR = {
						culture = culture:rimmen
						culture = { any_parent_culture_or_above = { this = culture:rimmen } }
					}
				}
				desc = elsweyr.0006.desc.rimmen
			}
			triggered_desc = {
				trigger = {
					OR = {
						culture = culture:rimmenese
						AND = {
							culture = { any_parent_culture_or_above = { this = culture:rimmenese } }
							NOT = { culture = culture:rimmen }
						}
					}
				}
				desc = elsweyr.0006.desc.rimmenese
			}
			desc = elsweyr.0006.desc.other_culture
		}
		#Religion
		first_valid = {
			# Doing it as Two Moons
			triggered_desc = {
				trigger = {
					faith = faith:twomoonsdance
				}
				desc = elsweyr.0006.desc.twomoonsdance
			}
			#Doing it as Many Paths
			triggered_desc = {
				trigger = {
					faith = faith:manypathsdance
				}
				desc = elsweyr.0006.desc.manypathsdance
			}
			#Doing it as New Moon
			triggered_desc = {
				trigger = {
					faith = faith:order_new_moon
				}
				desc = elsweyr.0006.desc.newmoondance
			}
		}
	}
	
	theme = war
	
	left_portrait = {
		character = root
		animation = personality_rational
	}
	
	immediate = {
		character:mane_001 = { save_scope_as = ridthar }
		character:anequina_darloc_brae = { save_scope_as = darloc }
		root = { save_scope_as = winner }
		scope:previous_holder = { save_scope_as = loser }
		
		### Get the HK of Elsweyr
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		title:e_elsweyr = {
			change_title_holder = {
				holder = root
				change = scope:change
			}
		}
		add_to_global_variable_list = {
			name = elsweyr_variables
			target = flag:restored_elsweyr
		}
		resolve_title_and_vassal_change = scope:change
		hidden_effect = {
			if = { #If Anequina/Pellitine/Mane is primary title then we apply all imperial things to new title and make it primary
				limit = {
					OR = {
						primary_title = title:e_anequina
						primary_title = title:e_pellitine
						has_title = title:d_mane
					}
				}
				set_primary_title_to = title:e_elsweyr
			}
			reunite_elsweyr_scripted_effect = yes
			#Check for Leyawiin
			if = {
				limit = { title:k_rimmen = { is_de_jure_liege_or_above_target = title:d_leyawiin }}
				title:d_leyawiin = { set_de_jure_liege_title = title:k_alabaster }
			}
			if = {
				limit = { title:k_alabaster = { is_de_jure_liege_or_above_target = title:d_nomore }}
				title:d_nomore = { set_de_jure_liege_title = title:k_rimmen }
			}
			if = {
				limit = { #If capital is in de jure Elsweyr, then make it de jure capital of Elsweyr
					title:e_elsweyr = { is_de_jure_liege_or_above_target = root.capital_county }
				}
				title:e_elsweyr = {
					set_capital_county = root.capital_county
				}
			}
		}
	}
	
	trigger = {
		OR = {
			AND = {
				scope:title = title:e_pellitine
				has_title = title:e_anequina
			}
			AND = {
				scope:title = title:e_anequina
				has_title = title:e_pellitine
			}
		}
	}

	option = {
		name = elsweyr.0006.a
		
		### Destroy Pellitine and Anequina
		destroy_title = title:e_pellitine
		destroy_title = title:e_anequina
	}
}

#Rim-Men events
namespace = ek_rimmen

ek_rimmen.0001 = {
	type = character_event
	title = ek_rimmen.0001.t
	content_source = dlc_ek
	desc = ek_rimmen.0001.desc
	theme = realm

	left_portrait = {
		character = root
		animation = thinking
	}

	right_portrait = {
		character = scope:refugee_leader
		animation = beg
	}

	immediate = {
		if = { #If Beloren-Kaie is alive, save him as leader of refugees for ruler of Rimmen
			limit = { 
				character:kaie_100 = { 
					is_alive = yes
					is_landed = no #Doesn't hold any land
					NOT = { any_held_title = { is_noble_family_title = yes } } #Doesn't have noble family title
				}
				completely_controls = title:c_hakoshae
			}
			character:kaie_100 = { save_scope_as = refugee_leader }
		}
		#Generate akaviri leader otheriwse
		else = {
			create_character = {
				age = { 20 50 }
				gender_female_chance = 50
				dynasty = generate
				faith = faith:court_dawnbeauty
				culture = culture:ilniviri
				location = root.capital_province

				after_creation = {
					ek_character_setup_effect = yes
				}

				save_scope_as = refugee_leader
			}
		}
	}
	option = { #Accept refugees
		name = ek_rimmen.0001.a
		if = { #If Hakoshae is controlled then Rim-Men settle there
			limit = { completely_controls = title:c_hakoshae }
			title:c_hakoshae = {
				set_county_culture = culture:rimmen
			}
			scope:refugee_leader = {
				get_title = title:c_hakoshae
			}
			title:c_rimmen = {
				set_county_culture = culture:rimcat
			}
		}
		else = {
			title:k_rimmen = {
				random_in_de_jure_hierarchy = { #County held by ruler of Rimmen, but not capital
					limit = {
						tier = tier_county
						holder = root
						NOT = { culture = culture:rimmen }
						NOT = { this = holder.capital_county }
					}
					save_scope_as = rim_men_county
				}
				scope:rim_men_county = {
					set_county_culture = culture:rimmen
				}
				scope:refugee_leader = {
					get_title = scope:rim_men_county
				}
			}
		}
		#else convert and grant 1 random county
		scope:refugee_leader = {
			#Rimmen culture formed
			set_culture = culture:rimmen
			hidden_effect = {
				add_character_flag = converted_culture_this_lifetime
			}
			#Culture and religion protected
			vassal_contract_set_obligation_level = {
				type = religious_rights
				level = 1
			}
			vassal_contract_set_obligation_level = {
				type = cultural_rights
				level = 1
			}
		}
		culture:rimmen = { #Get positive acceptance with new people
			change_cultural_acceptance = {
				target = root.culture
				value = 40
				desc = cultural_acceptance_historical_relations
			}
			culture_become_alive_effect = yes
		}
		culture:rimcat = { #Get positive acceptance with new people
			change_cultural_acceptance = {
				target = root.culture
				value = 20
				desc = cultural_acceptance_historical_relations
			}
			culture_become_alive_effect = yes
		}
		#Get gold
		add_gold = { value = medium_gold_value }
		#Some troops
		create_maa_regiment = {
			type = akaviri_dragon_hunters
			check_can_recruit = no
			size = 3
		}
		#Opinion with refugee leader
		reverse_add_opinion = {
			target = scope:refugee_leader
			modifier = respect_opinion
			opinion = 40
		}
		after = { #Convert courtiers to Rimcat
		scope:refugee_leader = {
			every_courtier_or_guest = {
				limit = {
					culture = culture:ilniviri
					is_ai = yes
				}
				set_culture = culture:rimmen
				hidden_effect = {
					add_character_flag = converted_culture_this_lifetime
				}
			}
		}
	}
		ai_chance = {
			base = 100
		}
	}
	option = { #Reject refugees, get dread and respect from Atterbus
		name = ek_rimmen.0001.b
		add_dread = 20
		reverse_add_opinion = {
			target = scope:attrebus
			modifier = respect_opinion
			opinion = 40
		}
		ai_chance = {
			base = 0
		}
	}
	#after = { #Convert courtiers to Rimmen
		#root = {
			#every_courtier_or_guest = {
				#limit = {
					#culture = culture:anequinan
					#is_ai = yes
				#}
				#set_culture = culture:rimcat
				#hidden_effect = {

				#}
